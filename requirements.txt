# Bike Computer — Python dependencies
# Install on Raspberry Pi with:  pip install -r requirements.txt

# Adafruit CircuitPython / Blinka hardware abstraction layer
adafruit-blinka

# ILI9341 TFT display driver
adafruit-circuitpython-rgb-display

# SHTC3 — temperature + humidity (Waveshare Sense HAT B)
adafruit-circuitpython-shtc3

# LPS22HB — barometric pressure (Waveshare Sense HAT B)
adafruit-circuitpython-lps2x

# TCS34725 — colour + lux (Waveshare Sense HAT B)
adafruit-circuitpython-tcs34725

# Image composition for the TFT framebuffer
pillow

# GPS UART
pyserial
pynmea2

# Raw I2C access (used for ICM20948 IMU driver)
smbus2

# GPIO control (touch IRQ pin)
RPi.GPIO
